select SALEDETIL1.ID, SALEDETIL1.GOODSID, INDUSTRY.INDUSTRY_CNAME, PRODUCTTYPE.PRODUCTTYPE_NAME, SALEDETIL1.PLATFORM_ID, BRAND.BRAND_NAME, PROVINCE.PROVINCE_NAME, SALEDETIL1.SHOPNAME, CITY.CITY_NAME, SALEDETIL1.TITLE, SALEDETIL1.PRICE, SALEDETIL1.DISCOUNT, SALEDETIL1.SALE_VOLUME, SALEDETIL1.SALE_AMOUNT, SALEDETIL1.COMMEN_NUM, SALEDETIL1.POSITIVE_COMMEN, SALEDETIL1.MODERATE_COMMEN, SALEDETIL1.NEGATIVE_COMMEN, SALEDETIL1.COLLECTION_NUM, SALEDETIL1.REPUTATION, SALEDETIL1.SHOPTYPE, SALEDETIL1.SHELVES_TIME, SALEDETIL1.YEAR, SALEDETIL1.MONTH, SALEDETIL1.DETAIL_URL, SALEDETIL1.ATTUBUTE_ONE, SALEDETIL1.ATTUBUTE_TWO, SALEDETIL1.CREATETIME

from SALEDETIL1

LEFT JOIN INDUSTRY on SALEDETIL1.INDUSTRY_ID = INDUSTRY.ID 
LEFT JOIN PRODUCTTYPE on SALEDETIL1.PRODUCTTYPE_ID = PRODUCTTYPE.ID
LEFT JOIN BRAND on  SALEDETIL1.BRAND_ID = BRAND.ID
LEFT JOIN PROVINCE on  SALEDETIL1.PROVINCE_ID = PROVINCE.ID
LEFT JOIN CITY on  SALEDETIL1.CITY_ID = CITY.ID

where SALEDETIL1.PRODUCTTYPE_ID = 28 and SALEDETIL1.PLATFORM_ID = 2
and SALEDETIL1.ID in (select min(SALEDETIL1.ID) from SALEDETIL1  group by SALEDETIL1.TITLE)
